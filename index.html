Paralelismo OpenMP

// Configurar 12 hilos para el procesamiento
omp_set_num_threads(12);
double t0 = omp_get_wtime();

// Cada hilo procesa imágenes independientes
#pragma omp parallel for schedule(dynamic) num_threads(12)
for (int i = 0; i < num_images; i++) {
    // Lectura independiente por hilo
    unsigned char* img = stbi_load(path, &w, &h, &c, 1);
    // Resize a 64x64
    stbir_resize_uint8_linear(img, w, h, 0,
                              resized, 64, 64, 0, 1);
    // Sobel → Flatten → Normalizar
    apply_sobel(resized, sobel, 64, 64);
    for (int p = 0; p < FLATTENED_SIZE; p++)
        X_data[i*FLATTENED_SIZE+p] = sobel[p] / 255.0f;
}
⚡ schedule(dynamic) balancea la carga entre hilos cuando las imágenes tienen distintos tamaños originales. Cada hilo opera sobre su propio buffer de memoria → sin condiciones de carrera.

PERSONA 3

Arquitectura MLP & Forward Pass

Red Neuronal Multicapa implementada desde cero en CUDA y Python

Arquitectura 4096 → 128 → 1

Z1 = X · W1ᵀ + b1 [N × 128]
A1 = ReLU(Z1) = max(0, Z1)
Z2 = A1 · W2ᵀ + b2 [N × 1]
A2 = σ(Z2) = 1 / (1 + e⁻ᶻ²)

Forward Pass — Kernel CUDA

// Kernel: Capa 1 (Entrada → Oculta + ReLU)
__global__ void forward_layer1(
    float* X, float* W1, float* b1,
    float* A1, int num_images)
{
    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;

    if (row < num_images && col < HIDDEN_SIZE) {
        float sum = 0.0f;
        for (int i = 0; i < INPUT_SIZE; i++)
            sum += X[row*INPUT_SIZE+i] * W1[col*INPUT_SIZE+i];
        sum += b1[col];
        // ReLU inline
        A1[row*HIDDEN_SIZE+col] = fmaxf(0.0f, sum);
    }
}

Forward Pass — Inferencia Python

# Pesos cargados desde binarios CUDA (.bin)
W1 = np.fromfile("weights/W1.bin",
     dtype=np.float32).reshape(128, 4096)

# Capa oculta: Z1 = X·W1ᵀ + b1 → A1 = ReLU
z1 = np.dot(x_vector, w1.T) + b1
a1 = np.maximum(0, z1)             # ReLU

# Capa salida: Z2 = A1·W2ᵀ + b2 → A2 = σ
z2 = np.dot(a1, w2.T) + b2
a2 = 1.0 / (1.0 + np.exp(-z2))   # Sigmoid

# Umbral de decisión
pred = "ENCORVADA" if a2[0] >= 0.5 else "RECTA"

Modelo de Ejecución CUDA

// Configuración dinámica: block 32×32
dim3 threadsPerBlock(32, 32);  // 1024 threads/block
dim3 grid_FW1(
  (HIDDEN_SIZE + 32 - 1) / 32,
  (num_images   + 32 - 1) / 32
);

// Lanzamiento del kernel
forward_layer1<<<grid_FW1, threadsPerBlock>>>(
    d_X, d_W1, d_b1, d_A1, N);

Backward Pass — Kernel Capa 1

__global__ void compute_dz1(
    float* dZ2, float* W2,
    float* A1,  float* dZ1, int N)
{
    int row = blockIdx.y*blockDim.y + threadIdx.y;
    int col = blockIdx.x*blockDim.x + threadIdx.x;
    if (row < N && col < HIDDEN_SIZE) {
        // Derivada de ReLU (función escalón)
        float relu_d = (A1[row*HIDDEN_SIZE+col] > 0.0f)
                       ? 1.0f : 0.0f;
        dZ1[row*HIDDEN_SIZE+col] =
            dZ2[row] * W2[col] * relu_d;
    }
}

Limitaciones, Trabajo Futuro & Conclusiones

Reflexión técnica sobre el pipeline implementado

Limitaciones Identificadas

📊
Dataset relativamente pequeño
1489 imágenes. Mayor diversidad mejoraría la generalización.
💡
Sensibilidad a condiciones de iluminación
Variaciones extremas de luz afectan la respuesta del filtro Sobel.
🧠
Limitaciones del MLP plano (4096→128→1)
No captura jerarquías espaciales como lo haría una CNN.

Trabajo Futuro

🔲 CNN — Capas convolucionales para extracción de características espaciales
🔄 Data Augmentation — Rotaciones, flips, ruido para ampliar dataset
⚡ Memoria Compartida CUDA — Shared memory para reducir accesos globales
🎯 cuBLAS — Librería NVIDIA para multiplicación de matrices altamente optimizada

Conclusiones Técnicas

01
OpenMP redujo significativamente los tiempos de preprocesamiento 12 hilos procesando imágenes independientes con schedule(dynamic) para balance de carga óptimo.
02
CUDA aceleró el entrenamiento neuronal mediante paralelismo masivo Kernels lanzados con grids 2D. Cada thread computa un elemento de la matriz de activaciones.
03
El filtro Sobel demostró ser efectivo como extractor de características Reduce 3-canal RGB a geometría corporal en 1 canal → entrada más limpia y compacta para el MLP.
04
Red neuronal implementada desde cero sin frameworks externos Forward + Backward pass, BCE Loss, L2 regularización y Early Stopping implementados en C/CUDA puro.
05
Pipeline completo funcional de extremo a extremo Imágenes brutas → Preprocesamiento → Entrenamiento GPU → Inferencia en tiempo real con Streamlit.
